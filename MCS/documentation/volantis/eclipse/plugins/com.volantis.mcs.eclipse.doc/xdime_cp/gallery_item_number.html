gallery:item-number

Purpose

Returns number of the item being displayed by Slideshow widget.

Contained by

gallery:slideshow

Attribute group

Core attributes

Example

<widget:popup id="gallery-popup" class="popup">
   <gallery:slideshow id="slideshow" class="slideshow">
      <gallery:item-display/>
      <table>
         <tr>
            <td>
               <widget:previous>&lt;&lt;</widget:previous>
            </td>
            <td>
               <widget:pause>Pause</widget:pause>
            </td>
            <td>
               <widget:play>Play</widget:play>
            </td>
            <td>
               <widget:next>&gt;&gt;</widget:next>
            </td>
            <td>
               <widget:dismiss>Close</widget:dismiss>
            </td>
         </tr>
      </table>
      <div class="slideshow-info">Displaying item <gallery:item-number/> of
         <gallery:items-count/></div>
   </gallery:slideshow>
</widget:popup>
