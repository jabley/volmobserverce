canvas

AttributeDescriptionTypeDefaultOptionsUse

Example

<table cols="2">
   <thead>
      <tr>
         <th>Attribute</th>
         <th>Description</th>
      </tr>
   </thead>
   <tfoot>
      <tr>
         <th>Attribute</th>
         <th>Description</th>
      </tr>
   </tfoot>
   <tbody keepTogether="true">
      <tr>
         <td>layoutName</td>
         <td>Name of the layout to use</td>
      </tr>
      ...
   </tbody>  
</table>
