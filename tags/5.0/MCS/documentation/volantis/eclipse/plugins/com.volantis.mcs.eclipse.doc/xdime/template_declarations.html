template:declarations

Purpose

Contains one or more parameter declarations.

Contains

template:parameter

Contained by

template:definition

Example

<template:definition>
   <template:declarations>
      <template:parameter name="page_title" type="complex"
         use="required" />
      <template:parameter name="page_content" type="complex"
         use="optional" />
      <template:parameter name="page_layout" type="simple"
         expressionVariable="PAGE_LAYOUT" use="required" />
      <template:parameter name="cache_key" type="simple" 
         expressionVariable="CACHE_KEY" use="required" />
   </template:declarations>
   ...
</template:definition>
