xf:item

Purpose

Container for the option values in xf:select and xf:select1 elements.

Contains

xf:label, xf:value

Contained by

xf:select, xf:select1

Attribute groups

Core attributes, Selection attributes

Example

<?xml version="1.0" encoding="UTF-8"?>
<html xmlns="http://www.w3.org/2002/06/xhtml2"
  xmlns:si="http://www.volantis.com/xmlns/2006/01/xdime2/si"
  xmlns:xf="http://www.w3.org/2002/xforms">
  <head>
    <title>xf:item</title>
    <xf:model id="selection">
      <xf:instance>
        <si:instance>
          <si:item name="list"/>
        </si:instance>
      </xf:instance>
    </xf:model>
  </head>
  <body>
    <xf:select1 model="selection" ref="list">
      <xf:label>Search by city</xf:label>
      <xf:item>
        <xf:label>London</xf:label>
        <xf:value>London</xf:value>
      </xf:item>
      <xf:item>
        <xf:label>Paris</xf:label>
        <xf:value>Paris</xf:value>
      </xf:item>
      <xf:item>
        <xf:label>Sydney</xf:label>
        <xf:value>Sydney</xf:value>
      </xf:item>
    </xf:select1>
  </body>
</html>
