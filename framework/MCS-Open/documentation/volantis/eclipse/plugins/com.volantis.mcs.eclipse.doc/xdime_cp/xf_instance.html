xf:instance

Purpose

Optional element that contains instance data associated with the model element. It is used to supply initial values for forms controls, or to provide additional data to be submitted with the form.

Contains

si:instance

Contained by

xf:model

Example

<?xml version="1.0" encoding="UTF-8"?>
<html xmlns="http://www.w3.org/2002/06/xhtml2"
  xmlns:si="http://www.volantis.com/xmlns/2006/01/xdime2/si"
  xmlns:xf="http://www.w3.org/2002/xforms">
  <head>
    <title>Subscriber update</title>
    <xf:model id="address">
      <xf:instance>
        <si:instance>
          <si:item name="name"/>
          <si:item name="postcode">13004 Marseille</si:item>
          <si:item name="hidden">subscriber_id</si:item>
        </si:instance>
      </xf:instance>
      <xf:submission id="submit" action="http://localhost:8080/volantis/response.jsp"/>
    </xf:model>
  </head>
  <body>
    <p>
      <xf:input model="address" ref="name">
        <xf:label>Name:</xf:label>
      </xf:input>
    </p>
    <p>
      <xf:input model="address" ref="postcode">
        <xf:label>Postcode:</xf:label>
      </xf:input>
    </p>
    <xf:submit submission="submit">
      <xf:label>Submit</xf:label>
    </xf:submit>
  </body>
</html>
