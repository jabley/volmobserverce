host

Purpose

The URI of the SMTP server.

Contained by

smtp

Example

<logging>
  <log-file>device_log</log-file>
  <e-mail>
    <e-mail-sending>enable</email-sending>
    <config>
      <smtp>
        <host>smtp.mycorp.com</host>
      </smtp>
      <from>
        <address>admin@mycorp.com</address>
        <name>mcs_admin</name>
      </from>
      <to>
        <address>device.team@volantis.com</address>
        <name>volantis_devices</name>
      </to>
      <period>week</period>
    </config>
  </e-mail>
</logging>
