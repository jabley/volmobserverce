ticker:item-title

Purpose

Provides the title of the item being displayed by the Item Display widget.

Contained by

ticker:feed, ticker:item-display

Attribute group

Core attributes

Example

<?xml version="1.0" encoding="UTF-8"?>
<html xmlns="http://www.w3.org/2002/06/xhtml2"
  xmlns:widget="http://www.volantis.com/xmlns/2006/05/widget"
  xmlns:ticker="http://www.volantis.com/xmlns/2006/09/ticker">
  <head>
    <title>ticker:item-title</title>
  </head>
  <body>
    <ticker:feed-poller
      url="http://localhost:8080/buzzcast/vmds/?uid=tickeruser&amp;delivery-format=4"
      style="min-polling-interval:120; max-polling-interval:120; "/>
    <widget:ticker-tape style="height:30px; width:300px; border: solid 2px green; ">
      <ticker:feed item-display="id"/>
    </widget:ticker-tape>
    <div>
      <ticker:item-display id="id" style="left:8px;top: 110px;height:260px; width:300px; border:
        solid 2px red; ">
        <div><strong><ticker:item-title /></strong></div>
        <div><ticker:item-description /></div>
        <div><ticker:item-channel /></div>
        <div><widget:dismiss>Close</widget:dismiss></div>
      </ticker:item-display>
    </div>
  </body>
</html>
