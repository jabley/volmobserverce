from

Purpose

The address of the sender of the log report.

Contains

address, name

Contained by

config

Example

<devices preload="false">
   ...
   <logging>
      <log-file>device_log</log-file>
      <e-mail>
         <e-mail-sending>enable</email-sending>
         <config><smtp><host>smtp.mycorp.com</host></smtp>
            <from>
               <address>admin@mycorp.com</address>
               <name>mcsAdmin</name>
            </from>
            <to>
               <address>device.team@volantis.com</address>
               <name>Devices</name>
            </to>
         </config>
      </e-mail>
   </logging>  
</devices>
