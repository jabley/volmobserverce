template:bindings

Purpose

Container for multiple template:binding elements, used with a full template definition.

Contains

template:binding

Example

<template:apply>
   <template:definition>
            ...
   </template:definition>
   <template:bindings>         
      <template:binding name="page_title">
         <template:complexValue>
	         <pipeline:try>
	            <pipeline:preferred>
                  <pipeline:transform href="title.xsl" compilable="true">
                     <urid:fetch href="title.xml" />
                  </pipeline:transform>	
               </pipeline:preferred>
               <pipeline:alternative>
                  <pipeline:content>MyCorp</pipeline:content>
               </pipeline:alternative>
            </pipeline:try>
         </template:complexValue>
      </template:binding>
      ...
   </template:bindings> 
</template:apply>
