wsdl-catalog

Purpose

Specifies alternative locations for WSDL documents.

Contains

wsdl-entry

Contained by

web-services-driver

Example

<web-services-driver>
   <wsdl-catalog>
      <wsdl-entry uri="uri/for/wsdl/entry">
         <class-resource path="/class/resource/path"/>
         <servlet-resource path="/servlet/resource/path"/>
         <uri-resource uri="uri/resource"/>
      </wsdl-entry>
   </wsdl-catalog>
</web-services-driver>
